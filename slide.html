  index.html:

    document.getElementById('show-screens-button').onclick = async function() {
      const screens = await navigator.screen.getScreens();
      console.log("getScreens returned " + screens.length + " screens:");
      for (const s of screens) {
        console.log(`'${s.name}' ${s.left},${s.top} ${s.width}x${s.height} ` +
                    `scaleFactor:${s.scaleFactor}, colorDepth:${s.colorDepth} ` +
                    `primary:${s.primary}, internal:${s.internal}`);
      }
    }

    document.getElementById('show-notification-button').onclick = function() {
      navigator.serviceWorker.ready.then(function(registration) {
        registration.showNotification('Click to open a window!');
      });
    }

  service-worker.js:

    self.addEventListener('notificationclick', async function(event) {
      event.notification.close();
      const screens = await navigator.screen.getScreens();
      var options = { x:      screens[0].left,
                      y:      screens[0].top,
                      width:  screens[0].width,
                      height: screens[0].height,
                      type:  "window" };
      clients.openWindow('https:wikipedia.org', options);
    });
